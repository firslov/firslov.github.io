1 Grafana setup

1.1 配置源

# 首先信任 https://packages.grafana.com/ 的 GPG 公钥:
curl https://packages.grafana.com/gpg.key | sudo apt-key add -
# 确保你的 apt 支持 HTTPS:
sudo apt-get install -y apt-transport-https

根据版本，将下面内容写进 /etc/apt/sources.list.d/grafana.list 

# stable
deb https://mirrors.tuna.tsinghua.edu.cn/grafana/apt/ stable main
# beta
deb https://mirrors.tuna.tsinghua.edu.cn/grafana/apt/ beta main

1.2 安装

sudo apt-get update
sudo apt-get install grafana

1.3 启动服务

配置文件位置： /etc/grafana/grafana.ini 

systemctl start grafana-server
systemctl status grafana-server
# 开机自启
sudo systemctl enable grafana-server.service

1.4 访问

浏览器访问 http://localhost:3000/ 如果您尚未配置其他端口，则 3000 是 Grafana 侦听的默认 http 端口。

2 InfluxDB setup

2.1 安装

sudo apt install influxdb
sudo apt install influxdb-client

2.2 配置

查看状态： 

sudo service influxdb status

开机启动： 

sudo service influxdb start

2.3 常用操作

启动： 

influxd

进入命令行： 

influx

数据库操作： 

# 查看数据库
show databases
# 创建数据库
create database testDb
# 选择数据库
use testDb

表操作： 

# 查看当前数据库下全部表
show measurements
# 创建"表"，直接在插入数据的时候指定表名，test 即表名，其中 host, monitor_name 是标签，count 是 field
insert test,host=127.0.0.1,monitor_name=test count=1
# 查看数据
select * from test
# 删除"表"
drop measurement "measurement_name"

4 Test demo

用于测试数据发送，请先调试好 grafana 的数据库接口。 

import datetime
import random
import time
from influxdb import InfluxDBClient

def send_data(client, i):
    data = [[27.33, 0.53, 68.19, 2.03, 1.74, 0.36, 941.09, 125.04, 23.27, 3.14, 44.08, 8.74],
            [27.65, 0.58, 68.11, 3.99, 1.94, 0.56, 1106.82, 118.02, 19.02, 1.38, 35.41, 6.23],
            [28.50, 0.62, 67.00, 4.51, 2.14, 0.75, 1326.59, 234.81, 23.64, 3.27, 49.66, 10.73]]
    point = ["L15", "L455", "L705"]
    current_time = datetime.datetime.utcnow().isoformat("T")
    body = [
        {
            "measurement": point[i],
            "time": current_time,
            "tags": {
                "point": point[i]
            },
            "fields": {
                "temperature": data[i][0]+data[i][1]*random.uniform(-1,1),
                "hunmidity": data[i][2]+data[i][3]*random.uniform(-1,1),
                "nh3": data[i][4]+data[i][5]*random.uniform(-1,1),
                "co2": data[i][6]+data[i][7]*random.uniform(-1,1),
                "pm2.5": data[i][8]+data[i][9]*random.uniform(-1,1),
                "pm10": data[i][10]+data[i][11]*random.uniform(-1,1)
            }
        }
    ]
    res = client.write_points(body)

def main():
    client = InfluxDBClient('127.0.0.1', 8086, 'username', 'passwd', 'dashboard')
    while True:
        for i in range(3):send_data(client, i)
        time.sleep(60)

if __name__ == "__main__":
    main()
